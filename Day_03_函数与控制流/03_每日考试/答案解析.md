public fun generate_pascals_triangle(n: u64): vector<vector<u64>> {
    let mut triangle = vector::empty();
    let mut row_num = 0;
    
    while (row_num < n) {
        let mut row = vector::empty();
        let mut col = 0;
        
        while (col <= row_num) {
            if (col == 0 || col == row_num) {
                vector::push_back(&mut row, 1);
            } else {
                let prev_row = vector::borrow(&triangle, row_num - 1);
                let left = *vector::borrow(prev_row, col - 1);
                let right = *vector::borrow(prev_row, col);
                vector::push_back(&mut row, left + right);
            };
            col = col + 1;
        };
        
        vector::push_back(&mut triangle, row);
        row_num = row_num + 1;
    };
    
    triangle
}

public fun bubble_sort(vec: &mut vector<u64>) {
    let len = vector::length(vec);
    let mut i = 0;
    
    while (i < len) {
        let mut j = 0;
        while (j < len - i - 1) {
            if (*vector::borrow(vec, j) > *vector::borrow(vec, j + 1)) {
                swap(vec, j, j + 1);
            };
            j = j + 1;
        };
        i = i + 1;
    };
}

public fun binary_search(vec: &vector<u64>, target: u64): u64 {
    let mut left = 0;
    let mut right = vector::length(vec);
    
    while (left < right) {
        let mid = left + (right - left) / 2;
        let mid_val = *vector::borrow(vec, mid);
        
        if (mid_val == target) {
            return mid
        } else if (mid_val < target) {
            left = mid + 1;
        } else {
            right = mid;
        };
    };

    let mut result = vector::empty();
    let len1 = vector::length(vec1);
    let len2 = vector::length(vec2);

    let mut j = 0;
    
    while (i < len1 && j < len2) {
        if (*vector::borrow(vec1, i) <= *vector::borrow(vec2, j)) {
            vector::push_back(&mut result, *vector::borrow(vec1, i));
            i = i + 1;

            j = j + 1;

        vector::push_back(&mut result, *vector::borrow(vec1, i));

    };
