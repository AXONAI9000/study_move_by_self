module my_addr::voting {
    use std::signer;
    use std::string::{Self, String};
    use std::vector;

    struct Proposal has key {
        title: String,
        yes_votes: u64,
        no_votes: u64,
        voters: vector<address>
    }

    const E_PROPOSAL_ALREADY_EXISTS: u64 = 1;
    const E_PROPOSAL_NOT_FOUND: u64 = 2;
    const E_ALREADY_VOTED: u64 = 3;

    public fun create_proposal(account: &signer, title: vector<u8>) {
        let addr = signer::address_of(account);
        assert!(!exists<Proposal>(addr), E_PROPOSAL_ALREADY_EXISTS);
        
        move_to(account, Proposal {
            title: string::utf8(title),
            yes_votes: 0,
            no_votes: 0,
            voters: vector::empty()
        });
    }

    public fun vote_yes(voter: &signer, proposal_addr: address) acquires Proposal {
        let voter_addr = signer::address_of(voter);
        let proposal = borrow_global_mut<Proposal>(proposal_addr);
        
        assert!(!vector::contains(&proposal.voters, &voter_addr), E_ALREADY_VOTED);
        
        proposal.yes_votes = proposal.yes_votes + 1;
        vector::push_back(&mut proposal.voters, voter_addr);
    }

    public fun vote_no(voter: &signer, proposal_addr: address) acquires Proposal {
        let voter_addr = signer::address_of(voter);
        let proposal = borrow_global_mut<Proposal>(proposal_addr);
        
        assert!(!vector::contains(&proposal.voters, &voter_addr), E_ALREADY_VOTED);
        
        proposal.no_votes = proposal.no_votes + 1;
        vector::push_back(&mut proposal.voters, voter_addr);
    }

    public fun get_votes(addr: address): (u64, u64) acquires Proposal {
        let proposal = borrow_global<Proposal>(addr);
        (proposal.yes_votes, proposal.no_votes)
    }

    public fun has_voted(proposal_addr: address, voter: address): bool acquires Proposal {
        let proposal = borrow_global<Proposal>(proposal_addr);
        vector::contains(&proposal.voters, &voter)
    }
}
